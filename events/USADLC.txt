# The Louisiana Purchase
country_event = {
	id = usa_dlc.65
	title = "usa_dlc.EVTNAME65"
	desc = "usa_dlc.EVTDESC65"
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
	
	trigger = {
		has_dlc = "American Dream"
		OR = {
			tag = USA
			has_country_flag = usa_declaration_independence_flag
		}
		BFR = {	ai = yes }
		NPF = {	ai = yes }
		colonial_louisiana = {
			OR = {
				owned_by = BFR
				owned_by = NPF
				owner = {
					is_colonial_nation_of = BFR
				}
				owner = {
					is_colonial_nation_of = NPF
				}
			}
			any_neighbor_province = {
				owned_by = root
			}
		}
	}
	
	mean_time_to_happen = { months = 200 }
	
	option = {
		name = "usa_dlc.EVTOPTA65" # Accept Offer
		add_treasury = -1000
		if = {
            limit = {
                exists = BFR
            }
			add_opinion = { 
			    who = BFR 
			    modifier = opinion_louisiana
			    years = 50
			}
            colonial_louisiana = {
                if = {
                    limit = {
                        OR = {
                            owned_by = BFR
                            owner = { is_colonial_nation_of = BFR }
                        }
                    }
                    owner = {
                        remove_core = PREV
                    }
                    add_core = ROOT
                    cede_province = ROOT
                }
            }
        }
		if = {
            limit = {
                exists = NPF
            }
			add_opinion = { 
			    who = NPF 
			    modifier = opinion_louisiana
			    years = 50
			}
            colonial_louisiana = {
                if = {
                    limit = {
                        OR = {
                            owned_by = NPF
                            owner = { is_colonial_nation_of = NPF }
                        }
                    }
                    owner = {
                        remove_core = PREV
                    }
                    add_core = ROOT
                    cede_province = ROOT
                }
            }
        }
		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					is_bankrupt = yes
					is_in_large_debt = yes
					AND = {
						NOT = { treasury = 500 }
						NOT = { monthly_income = 10 }
					}
					AND = {
						NOT = { treasury = 250 }
						NOT = { monthly_income = 20 }
					}
				}
			}
		}
	}
	
	option = { 
        name = "usa_dlc.EVTOPTB65" # Decline Offer
        if = {
		    limit = {
                exists = BFR
            }
            add_opinion = { 
			    who = BFR 
			    modifier = opinion_disappointed
			    years = 20
			}
        }
        if = {
		    limit = {
                exists = NPF
            }
            add_opinion = {
                who = NPF 
			    modifier = opinion_disappointed
			    years = 20
            }
        }
		ai_chance = {
			factor = 1
		}
    }
}
